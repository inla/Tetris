6   package tetris.ui;
7   
8   import java.awt.Color;
9   import java.awt.Graphics;
10  import javax.swing.JPanel;
11  import tetris.logic.Game;
12  import tetris.logic.Tetrimino;
13  
14  /**
15   * Panel that shows score and next tetrimino.
16   *
17   * @author inka
18   */
19  public class SidePanel extends JPanel {
20  
21      private final int squareSize;
22      private Game game;
23      private Tetrimino tetrimino;
24      private int score;
25      private int removedRows;
26      private int level;
27  
28      public SidePanel(Game game, int squareSize) {
29          super.setBackground(Color.BLACK);
30          this.squareSize = squareSize;
31          this.game = game;
32          this.tetrimino = this.game.getNextTetrimino();
33          this.score = this.game.getScore();
34          this.removedRows = this.game.getRemovedRows();
35          this.level = this.game.getLevel();
36      }
37  
38      @Override
39      protected void paintComponent(Graphics g) {
40          super.paintComponent(g);
41          paintTetrimino(g);
42      }
43  
44      private void paintTetrimino(Graphics g) {
45          int[][] tetr = this.tetrimino.getCurrentRotation();
46          for (int i = 0; i < tetr.length; i++) {
47              for (int j = 0; j < tetr[i].length; j++) {
48                  if (tetr[i][j] != 0) {
49                      int x = this.tetrimino.getX() + i;
50                      int y = this.tetrimino.getY() + j;
51                      Color c = this.tetrimino.getColor();
52                      paintPoint(g, x, y, c);
53                  }
54              }
55          }
56      }
57  
58      private void paintPoint(Graphics g, int x, int y, Color c) {
59          g.setColor(c);
60          g.fill3DRect(x * squareSize, y * squareSize, squareSize, squareSize, true);
61      }
62  
63      @Override
64      public void repaint() {
65          super.repaint();
66      }
67  
68  }
