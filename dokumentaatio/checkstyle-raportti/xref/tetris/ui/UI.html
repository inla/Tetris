1   package tetris.ui;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import javax.swing.JFrame;
7   import javax.swing.WindowConstants;
8   import tetris.tetris.Game;
9   
10  /**
11   * The user interface of tetris.
12   *
13   * @author inka
14   */
15  public class UI implements Runnable {
16  
17      private static final int SQUARE_SIZE = 24;
18      private JFrame frame;
19      private final Game game;
20      private AbstractPanel gamePanel;
21      private AbstractPanel sidePanel;
22  
23      /**
24       * Creates the user interface.
25       *
26       * @param game the logic that needs user interface
27       */
28      public UI(Game game) {
29          this.game = game;
30      }
31  
32      @Override
33      public void run() {
34          frame = new JFrame("Tetris");
35          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
36          frame.setLayout(new BorderLayout(0, SQUARE_SIZE));
37  
38          createComponents(frame.getContentPane());
39          setPanelSizes();
40          frame.pack();
41          frame.setVisible(true);
42          frame.setResizable(false);
43      }
44  
45      private void createComponents(Container c) {
46          this.gamePanel = new GamePanel(this.game, SQUARE_SIZE);
47          this.sidePanel = new SidePanel(this.game, SQUARE_SIZE);
48          c.add(this.gamePanel, BorderLayout.CENTER);
49          c.add(this.sidePanel, BorderLayout.EAST);
50          frame.addKeyListener(new KeyboardListener(this.game));
51      }
52  
53      private void setPanelSizes() {
54          int width = SQUARE_SIZE * this.game.getBoard().getWidth();
55          int height = SQUARE_SIZE * this.game.getBoard().getHeight();
56          this.gamePanel.setPreferredSize(new Dimension(width, height));
57          this.sidePanel.setPreferredSize(new Dimension(width - SQUARE_SIZE, height));
58      }
59  
60      public AbstractPanel getGamePanel() {
61          return gamePanel;
62      }
63  
64      public AbstractPanel getSidePanel() {
65          return sidePanel;
66      }
67  
68      public JFrame getFrame() {
69          return frame;
70      }
71  
72  }
