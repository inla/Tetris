1   package tetris.ui;
2   
3   import com.sun.javafx.scene.traversal.Direction;
4   import java.awt.event.KeyEvent;
5   import java.awt.event.KeyListener;
6   import tetris.logic.Game;
7   
8   /**
9    * Handles the presses of the keyboard made by the player.
10   *
11   * @author inka
12   */
13  public class KeyboardListener implements KeyListener {
14  
15      private final Game game;
16  
17      public KeyboardListener(Game game) {
18          this.game = game;
19      }
20  
21      @Override
22      public void keyTyped(KeyEvent e) {
23      }
24  
25      @Override
26      public void keyPressed(KeyEvent e) {
27          if (e.getKeyCode() == KeyEvent.VK_UP) {
28              this.game.moveTetrimino(Direction.UP);
29          } else if (e.getKeyCode() == KeyEvent.VK_RIGHT) {
30              this.game.moveTetrimino(Direction.RIGHT);
31          } else if (e.getKeyCode() == KeyEvent.VK_DOWN) {
32              this.game.moveTetrimino(Direction.DOWN);
33          } else if (e.getKeyCode() == KeyEvent.VK_LEFT) {
34              this.game.moveTetrimino(Direction.LEFT);
35          } else if (e.getKeyCode() == KeyEvent.VK_SPACE) {
36              this.game.moveTetrimino(null);
37          } else if (e.getKeyCode() == KeyEvent.VK_P) {
38              this.game.pauseGame();
39          } else if (e.getKeyCode() == KeyEvent.VK_F1) {
40              this.game.initialize();
41              this.game.restart();
42          }
43          this.game.repaint();
44      }
45  
46      @Override
47      public void keyReleased(KeyEvent e) {
48      }
49  
50  }
