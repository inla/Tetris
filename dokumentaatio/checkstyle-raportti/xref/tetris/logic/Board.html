1   package tetris.logic;

3   import java.awt.Color;
4   
5   /**
6    * Represents the game board of Tetris. It is defined as a 2-dimensional integer
7    * table, where 0 means an empty point.
8    *
9    * @author inka
10   */
11  public class Board {
12  
13      private final static int HEIGHT = 20;
14      private final static int WIDTH = 10;
15      private final int[][] board;
16  
17      /**
18       * Creates a new board matrix.
19       */
20      public Board() {
21          this.board = new int[HEIGHT][WIDTH];
22      }
23  
24      public int getHeight() {
25          return HEIGHT;
26      }
27  
28      public int getWidth() {
29          return WIDTH;
30      }
31  
32      public int[][] getBoard() {
33          return board;
34      }
35  
36      /**
37       * Clears the board. Sets all points to 0.
38       */
39      public void emptyBoard() {
40          for (int i = 0; i < HEIGHT; i++) {
41              for (int j = 0; j < WIDTH; j++) {
42                  this.board[i][j] = 0;
43              }
44          }
45      }
46  
47      /**
48       * Adds a tetrimino into the board. Sets the coordinate values where the
49       * tetrimino is to the tetrimino's type (which is the colorCode).
50       *
51       * @param tetrimino the tetrimino to be added
52       */
53      public void addTetrimino(Tetrimino tetrimino) {
54          for (int i = 0; i < tetrimino.getCurrentRotation().length; i++) {
55              for (int j = 0; j < tetrimino.getCurrentRotation()[i].length; j++) {
56                  int colorCode = tetrimino.getCurrentRotation()[i][j];
57                  if (colorCode == 0) {
58                      continue;
59                  }
60                  int x = j + tetrimino.getX();
61                  int y = i + tetrimino.getY();
62                  setPoint(x, y, colorCode);
63              }
64          }
65      }
66  
67      /**
68       * Tells if a point in given coordinates is empty (=point is zero).
69       *
70       * @param x x coordinate
71       * @param y y coordinate
72       * @return true if the point is empty or above the board, false otherwise
73       */
74      public boolean isPointEmpty(int x, int y) {
75          if (x < 0 || x >= WIDTH || y < 0 || y >= HEIGHT) {
76              return x >= 0 && x < WIDTH && y < 0;
77          } else {
78              return this.board[y][x] == 0;
79          }
80      }
81  
82      /**
83       * Removes all full rows from the board. Iterates through all the rows from
84       * below upwards, checks if a row is full and if it is, removes it.
85       *
86       * @return how many rows were removed
87       */
88      public int removeFullRows() {
89          int removed = 0;
90          for (int y = HEIGHT - 1; y >= 0; y--) {
91              if (isRowFull(y)) {
92                  removeRow(y);
93                  removed++;
94                  y++;
95              }
96          }
97          return removed;
98      }
99  
100     /**
101      * Tells if a certain row is full. A row is full when all of its points are
102      * not empty.
103      *
104      * @param y row's y coordinate
105      * @return true if the row is full, false otherwise
106      */
107     public boolean isRowFull(int y) {
108         if (y < 0 || y >= HEIGHT) {
109             return false;
110         }
111         for (int x = 0; x < WIDTH; x++) {
112             if (this.board[y][x] == 0) {
113                 return false;
114             }
115         }
116         return true;
117     }
118 
119     /**
120      * Removes the given row. Drops the upper rows one row further down and sets
121      * the top row empty.
122      *
123      * @param y the y coordinate of the row to be removed
124      */
125     public void removeRow(int y) {
126         if (y < 0 || y >= HEIGHT) {
127             return;
128         }
129         for (int z = y; z > 0; z--) {       //ylemmät rivit yksi alemmas
130             for (int x = 0; x < WIDTH; x++) {
131                 this.board[z][x] = this.board[z - 1][x];
132             }
133         }
134         for (int x = 0; x < WIDTH; x++) {     //uusi rivi ylös
135             this.board[0][x] = 0;
136         }
137     }
138 
139     /**
140      * Sets the value of a given coordinate point to the given type.
141      *
142      * @param x x coordinate
143      * @param y y coordinate
144      * @param type type to be set (=colorcode)
145      */
146     public void setPoint(int x, int y, int type) {
147         if (x >= 0 && x < WIDTH & y >= 0 & y < HEIGHT) {
148             this.board[y][x] = type;
149         }
150     }
151 
152     /**
153      * Tells the type of a point in given coordinates.
154      *
155      * @param x point's x coordinate
156      * @param y point's y coordinate
157      * @return the type(=colorcode)
158      */
159     public int getPoint(int x, int y) {
160         if (x < 0 || x >= WIDTH || y < 0 || y >= HEIGHT) {
161             return 0;
162         } else {
163             return this.board[y][x];
164         }
165     }
166 
167     /**
168      * Tells the color of the point.
169      *
170      * @param x point's x coordinate
171      * @param y point's y coordinate
172      * @return color of the point
173      */
174     public Color getPointColor(int x, int y) {
175         int colorCode = getPoint(x, y);
176         for (TetriminoType type : TetriminoType.values()) {
177             if (type.getType() == colorCode) {
178                 return type.getColor();
179             }
180         }
181         return null;
182     }
183 }
